题目：812. 最大三角形面积

难度：简单

给定包含多个点的集合，从其中取三个点组成三角形，返回能组成的最大三角形的面积

示例 1:

输入: points = [[0,0],[0,1],[1,0],[0,2],[2,0]]
输出: 2
解释: 这五个点如下图所示。组成的橙色三角形是最大的，面积为2。

解题思路

牺牲空间的时间保护法
枚举可能的三角形面积，然后比较出最大的三角形面积.(巨low的做法…???看了看题解真的除了暴力没有办法解决吗)
(len == 3可以和下面的合并，不过合并了之后会变慢很多!!还有不要用库函数Math.max(area,Math.abs()))越来越慢!
这可是O(Cn3)=C(n^3)的复杂度，一开始想了个O(n^2)的复杂度的，结果只能跑一半多数据，仔细想了想，还大学生呢，这不那啥吗

    public double largestTriangleArea(int[][] points) {
        double area = 0;
       // double maxArea = 0;
        int len = points.length;
        double[] areaList = new double[len*(len-1)*(len-2)/6];
        int a1,a2,b1,b2;
        int index = 0;
        if(len == 3){
            a1 = points[0][0]-points[1][0];
            a2 = points[0][0]-points[2][0];
            b1 = points[0][1]-points[1][1];
            b2 = points[0][1]-points[2][1];
            area = 0.5*Math.abs(a1*b2-a2*b1);
            return area;
        }
        for(int i=0;i<len-2;i++){
            for(int j=i+1;j<len-1;j++){
                for(int k=j+1;k<len;k++){
                    a1 = points[i][0]-points[j][0];
                    a2 = points[i][0]-points[k][0];
                    b1 = points[i][1]-points[j][1];
                    b2 = points[i][1]-points[k][1];
                    areaList[index++] = 0.5*Math.abs(a1*b2-a2*b1);
                    /*area = 0.5*Math.abs(a1*b2-a2*b1);
                    if(area > maxArea){
                        maxArea = area;
                    }*/
                    /*area = Math.max(area,0.5*Math.abs(a1*b2-a2*b1));*/
                }
            }
        }
        for(int i=0;i<len*(len-1)*(len-2)/6;i++){
            if(areaList[i] > area){
                area = areaList[i];
            }
        }
        return area;
    }
}
public class Solution {
  /*  public static void main(String[] args) {

        System.out.println(solution.largestTriangleArea(points));

/*
     * 1.找到距离最远的两点P1,P2
     * 2.找到距离直线P1P2最远的一点P3
     * 3.三角形P1P2P3即为所求三角形，该面积最大
     */

    public double largestTriangleArea(int[][] points) {
        int[] targets = new int[2];
        searchMaxLength(points,targets);
        int p3 = searchFarPoint(points,targets);
        int a1 = points[targets[0]][0]-points[p3][0];
        int a2 = points[targets[1]][0]-points[p3][0];
        int b1 = points[targets[0]][1]-points[p3][1];
        int b2 = points[targets[1]][1]-points[p3][1];
        return 0.5*Math.abs(a1*b2-a2*b1);
    }
    public void searchMaxLength(int[][] points,int[] targets){
        double maxLength = 0;
        double length = 0;
        for(int i=0;i<points.length-1;i++){
            for(int j=i+1;j<points.length;j++){
                length = Math.pow(points[i][0]-points[j][0],2)+Math.pow(points[i][1]-points[j][1],2);
                if(length > maxLength){
                    maxLength = length;
                    targets[0] = i;
                    targets[1] = j;
                }
            }
        }
    }
    public int searchFarPoint(int[][] points,int[] target){
        int p3 = 0;
        if(points[target[0]][0]-points[target[1]][0] == 0){
            int x = points[target[0]][0];
            int maxLength = -1;
            for(int i=0;i<points.length;i++){
                if(Math.abs(points[i][0] - x) > maxLength){
                    maxLength = Math.abs(points[i][0] - x);
                    p3 = i;
                }
            }
            return p3;
        }
        //当时这里用的时候有bug 因为 points的运算都是整形 比如3/10 = 0 然后把0赋给k 而不是0.3 所以用的时候
        //可以通过和0.0的运算进行浮点型和整形数据的转化
        double k = (points[target[0]][1]-points[target[1]][1]+0.0)/(points[target[0]][0]-points[target[1]][0]+0.0);
        double t = points[target[0]][1]-k*points[target[0]][0];
        double d = -1;
        for(int i=0;i<points.length;i++){
            if(Math.abs(k*points[i][0]-points[i][1]+t) > d){
                d = Math.abs(k*points[i][0]-points[i][1]+t);
                p3 = i;
            }
        }
        return p3;
    }
}
